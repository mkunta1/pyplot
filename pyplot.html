import random

plt.figure(figsize=(12, 5))


# Scatter plot
plt.subplot(1, 2, 1)
plt.scatter(x_values, random_numbers, color='brown', s=100, alpha=0.7)

plt.xlabel('Index')
plt.ylabel('Random Value')
